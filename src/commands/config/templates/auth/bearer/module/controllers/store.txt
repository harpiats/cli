import type { Request, Response } from "@harpia/core";
import { ApiResponse } from "@harpia/common";
import { service } from "../services";
import { validation } from "../validations";

export async function store(request: Request, response: Response) {
  try {
    const body = await request.json();
    await validation.create(body);

    const tokenPayload = await service.create(body);

    return ApiResponse.success(response, tokenPayload);
  } catch (error: any) {
    return ApiResponse.error(response, error);
  }
}
